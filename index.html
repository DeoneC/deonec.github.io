function renderDashboard(data) { const container = document.getElementById("dashboard"); container.innerHTML = ""; data.forEach(item => { const cls = item.change >= 0 ? "positive" : "negative"; // Format price as $0.00 const priceFormatted = `$${item.price.toLocaleString(undefined, { minimumFractionDigits: 2, maximumFractionDigits: 2 })}`; // Format percentage change (+1.23% or -0.45%) const changeFormatted = `${item.change >= 0 ? "+" : ""}${item.change.toFixed(2)}%`; const card = document.createElement("div"); card.className = "card"; card.innerHTML = `
${item.name}
${priceFormatted}
${changeFormatted}
`; container.appendChild(card); }); }
